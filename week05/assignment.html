adding group data to coldata of RangedSummarizedExperiment

rownames <- se@colData@rownames
condition <- sub("^(.*?)_(male|female).*", "\\1.\\2", rownames)
se$condition <- condition
sex <- sub("^(.*?)_(male|female).*", "\\2", rownames)
se$sex <- sex
stress <- condition <- sub("^(.*?)_(male|female).*", "\\1.", rownames)
se$stress <- stress

differential analysis

# create DGEList (format for edgeR)
dds_a <- edgeR::DGEList(assay(se), group=se$condition)
# normalization with TMM
dds_a <- edgeR::calcNormFactors(dds_a)

# set up matrix & compute dispersion
mm_b <- model.matrix(~ sex + stress, data=as.data.frame(colData(se)))
dds_b <- estimateDisp(dds_a, mm_b)
fit_b <- glmQLFit(dds_b, mm_b)
# add normalized data to RangedSummarizedExperiment object
assay(se, "logcpm") <- log1p(cpm(dds_b))

comparing stressed and control animals

# Test for effect of Stress (adjusted for sex)
res_stress <- as.data.frame(topTags(glmQLFTest(fit_b, coef="stressFSS."), n=10)) 

# get top10 most significant regions
topRegions_stress <- head(row.names(res_stress),10)
