download ATACseq data

download.file("https://www.ethz-ins.org/content/ATAC_AstroVsESC_subset.rds", "ATACseq.gz")
ATACseq <- readRDS("ATACseq.gz")

visualize comparison between DNA methylation and chromatin accessibility

used barplot idea from following paper https://pmc.ncbi.nlm.nih.gov/articles/PMC8897925/#Sec11

# add column defining if differentially accessibility is opening or closing
mcols(ATACseq)$category <- with(mcols(ATACseq), ifelse(
  FDR < 0.05 & logFC > 0, "Opening",
  ifelse(FDR < 0.05 & logFC < 0, "Closing", "Stable")
))

# add column defining if DMR regions have increased vs decreased methylation
DMR_sign <- DMR[DMR$HMFDR < 0.05]
DMR_sign$direction <- ifelse(DMR_sign$meandiff < 0,
                             "Decreased Methylation",
                             "Increased Methylation")
# find DAR that overlap DMRs & combine results in a dataframe
hits <- findOverlaps(DMR_sign, ATACseq)
df <- data.frame(
  DMR_id = queryHits(hits),
  direction = mcols(DMR_sign)$direction[queryHits(hits)],
  category = mcols(ATACseq)$category[subjectHits(hits)]
)
# if one DMR is overlaped by multiple DARs, select accessibility category that appears the most in that DMR
df_dmr <- df %>%
  group_by(DMR_id, direction) %>%
  summarize(
    category = names(sort(table(category), decreasing = TRUE))[1],
    .groups = "drop"
  )

#make table & add percent
plot_df <- df_dmr %>%
  count(direction, category)
plot_df <- plot_df %>%
  group_by(direction) %>%
  mutate(percent = n / sum(n))

# plot results as barplot
ggplot(plot_df, aes(x = direction, y = percent, fill = category)) +
  geom_bar(stat = "identity", width = 0.6) +
  
  # counts inside bars
  geom_text(aes(label = n),
            position = position_stack(vjust = 0.5),
            size = 4) +
  
  scale_y_continuous(labels = scales::percent_format()) +
  
  labs(
    x = NULL,
    y = "% of Differentially Methylated Regions 
    that overlap types of differential chromatin
    accessibility Regions",
    fill = "accesibility of chromatin"
  ) +
  
  theme_classic() +
  theme(legend.position = "right")

The plot shows that in DMR regions with decreased methylation have a higher proportion of opening chromatin/accessible DNA, while DMR with increased methylation show the opposite: more closing chromatin. This relationship is not surprising, however I expected it to be even stronger. Also there’s much more regions with decreased methylation compared to increased methylation
